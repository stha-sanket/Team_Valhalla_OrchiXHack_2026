\section*{User}
\begin{longtable}{@{}llp{0.44\textwidth}@{}}
  \toprule \textbf{Field} & \textbf{Type} & \textbf{Notes} \\ \midrule
  \code{name} & string & Display name \\
  \code{email} & string & Unique; login identity \\
  \code{password} & string & bcrypt hash \\
  \code{role} & string & \code{user} (default) or \code{admin} \\
  \code{milestones[]} & array & Permanent \code{\{name, earned\_at\}} log; survives resets \\
  \code{avatar} & string & Image URL \\
  \code{isVerified} / \code{isActive} & boolean & Email verified; soft-disable flag \\
  \code{*Token(+Expiry)} & string/Date & Verify, reset, and refresh token state \\
  \bottomrule
\end{longtable}

\begin{longtable}{@{}llp{0.44\textwidth}@{}}

\section*{Quiz and QuizAttempt}
\begin{longtable}{@{}llp{0.44\textwidth}@{}}
  \toprule \textbf{Field} & \textbf{Type} & \textbf{Notes} \\ \midrule
  \code{Quiz.visiting\_place\_id} & string & Unique --- one quiz per place \\
  \code{Quiz.questions[]} & array & \code{\{question, options[], correct\_index\}}; exactly five; answers never sent to users \\

  \code{QuizAttempt.correct\_count} / \code{points} & number & Score and award \\

\section*{Admin, Session, Role}
\code{Admin} mirrors \code{User} (name, email, bcrypt password, activity
flags) in a separate collection with a \code{permissions[]} array;
\code{Session} records \code{\{userId, token, ip, userAgent, expiresAt,
isActive\}} for multi-device management; \code{Role} defines named
permission sets \code{\{name, description, permissions[]\}}.
